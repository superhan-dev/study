테스트 문제

```javascript
const fs = require("fs");
function main() {
  const input = fs.readFileSync(0, "utf-8").trimEnd().split("\n");

  const map = {
    "-": 0,
    "'": 1,
    "(": 2,
    "@": 3,
    "?": 4,
    ">": 5,
    "&": 6,
    "%": 7,
    "/": -1,
  };

  let idx = 0;
  while (idx < input.length) {
    const line = input[idx];
    if (line === "#") break;

    const letters = line.split("");

    let power = letters.length - idx - 1;

    let sum = 0;
    for (let i = 0; i < letters.length; i++) {
      sum += map[letters[i]] * Math.pow(8, power);
    }

    console.log(sum);
    idx++;
  }
}
main();
```
